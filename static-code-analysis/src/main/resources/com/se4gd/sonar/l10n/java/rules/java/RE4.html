Do not access global variables inside for-loop; instead try to use local variables.

Non-compliant Code Examples

    public static int a = 6;
    
    public void test() {
        for(int i = 0; i < 5; i++) {
            System.out.println("The number is " + ClassName.a);
        }
        int i = 0;
        while(i < 5){
            System.out.println("The number is " + ClassName.a);
            i++;
        }
    }

Compliant Solutions

    public static int a = 6;
    
    public void test() {
        int b = 6;
        for(int i = 0; i < 5; i++) {
            System.out.println("The number is " + b);
        }
        int i = 0;
        while(i < 5){
            System.out.println("The number is " + b);
            i++;
        }
    }
